window.addEventListener("DOMContentLoaded", function(){
  rwd.picture(document.getElementsByTagName('picture'), {
    before: function() {
      console.log('before')
    },
    after: function() {
      console.log('after');
    },
    loadonce: function() {
      console.log('load once');
    },
    load: function() {
      console.log('load')
    }
  });
});
