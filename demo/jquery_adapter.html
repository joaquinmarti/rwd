$(document).ready(function() {
  $('picture').picture({
    before: function() {
      console.log('before 1');
    },
    after: function() {
      console.log('after 1');
    },
    loadonce: function() {
      console.log('load once 1');
    },
    load: function() {
      console.log('load 1');
    }
  });

});
